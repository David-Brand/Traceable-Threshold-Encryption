cmake_minimum_required(VERSION 3.12)
project(FingerprintingCodesTest CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -O2")
add_compile_options(/openmp)


add_library(fpcodes
    ../src/fingerprinting_codes.cpp
)

target_include_directories(fpcodes PUBLIC ../src)

add_executable(test_fpcodes test_fpcodes.cpp)
target_link_libraries(test_fpcodes PRIVATE fpcodes)
